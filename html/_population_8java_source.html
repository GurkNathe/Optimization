148 public double[][] genNeighborhood(int n, int m, double[] solution) {
149 double[][] neighborhood = new double[n][m];
150 MTRandom r = new MTRandom(false);
151
152 // Fill a neighborhood with pseudo-random values based off of the solution
153 // vector within a given range
154 for (int i = 0; i < n; i++) {
155 for (int j = 0; j < m; j++) {
156 double value = solution[j] + r.nextDouble() * (range - (-range)) + (-range);
157 if (value > range) {
158 value = range;
159 } else if (value < -range) {
160 value = -range;
161 }
162
163 neighborhood[i][j] = value;
164 }
165 }
166 return neighborhood;
