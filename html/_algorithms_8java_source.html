99 public double[] blindSearch(int iterations, double[] bestSol, double fitness) {
100 for (int i = 0; i < iterations; i++) {
101 // Gets a random solution
102 double[] arg = population.genRandomArray(m);
103
104 // Evaluate the fitness of the random solution
105 Problem prob = new Problem(arg, problem);
106
107 // Get the fitness of the solution
108 double fitnessNew = prob.getFitness();
109
110 // If the fitness is better than the current best solution,
111 // update the best solution
112 if (fitnessNew < fitness) {
113 fitness = fitnessNew;
114 bestSol = arg;
115 }
116 }
117 this.fitness = fitness;
118 return bestSol;

130 public double[] localSearch(double[] initialSol, double[] bestSol, boolean tau) {
131 // Run algorithm until the solution doesn't improve
132 while (tau) {
133 tau = false;
134
135 // Generate the neighborhood of the current solution
136 double[][] neighborhood = population.genNeighborhood(n, m, initialSol);
137
138 // Initialize best neighborhood fitness and solution
139 double bestFitnessInNeighborhood = Double.MAX_VALUE;
140 double[] bestSolutionInNeighborhood = new double[m];
141
142 // Evaluate the fitness of the neighborhood
143 for (int i = 0; i < neighborhood.length; i++) {
144 Problem prob = new Problem(neighborhood[i], problem);
145 double fitnessNew = prob.getFitness();
146
147 // If the fitness is better than the current best solution in neighborhood,
148 // update the best solution in neighborhood
149 if (fitnessNew <= this.fitness) {
150 bestFitnessInNeighborhood = fitnessNew;
151 bestSolutionInNeighborhood = neighborhood[i];
152 }
153 }
154
155 // If the best solution in the neighborhood is better than the current best
156 // solution, update the best solution
157 if (bestFitnessInNeighborhood < this.fitness) {
158 this.fitness = bestFitnessInNeighborhood;
159 bestSol = bestSolutionInNeighborhood;
160 tau = true;
161 }
162 }
163 return bestSol;

177 public double[] repeatedLocalSearch(double[] initialSol, double[] bestGlobalSol, double[] bestIterSol, boolean tau,
178 int iterations) {
179
180 // Initialize best solutions
181 bestGlobalSol = initialSol;
182 bestIterSol = initialSol;
183 tau = true;
184
185 // Iteration counter
186 int t = 1;
187
188 // Run the algorithm until all iterations are complete
189 while (t <= iterations) {
190 // Get the local search solution for current iteration
191 bestIterSol = localSearch(bestGlobalSol, bestIterSol, tau);
192
193 // Get fitness of current iteration and global iteration
194 Problem prob = new Problem(bestIterSol, problem);
195 Problem probGlobal = new Problem(bestGlobalSol, problem);
196
197 // If the fitness of the current iteration is better than the global
198 // solution, update the global solution
199 if (prob.getFitness() < probGlobal.getFitness()) {
200 bestGlobalSol = bestIterSol;
201 }
202 t++;
203 bestIterSol = population.genRandomArray(m);
204 }
205 return bestGlobalSol;
206 }

220 public double[] DE(int method, int D, int NP, double CR, double F, double lambda, int generations) {
221 int generation = 0;
222 while (generation < generations) {
223 // Iterate over every solution in the population
224 for (int i = 0; i < NP; i++) {
225
226 // Getting the randomly selected vector indexes
227 int r1 = i, r2 = i, r3 = i, r4 = i, r5 = i, jrand = r.nextInt(D);
228 while (same(i, r1, r2, r3, r4, r5)) {
229 r1 = r.nextInt(NP);
230 r2 = r.nextInt(NP);
231 r3 = r.nextInt(NP);
232 r4 = r.nextInt(NP);
233 r5 = r.nextInt(NP);
234 }
235
236 // noisy vector
237 double[] u = new double[D];
238
239 // Mutation of noisy vector
240 boolean crossed = false;
241 while (!crossed) {
242 for (int k = 0; k < D; k++) {
243 if (r.nextDouble() < CR || k == jrand) {
244 u[k] = method(method, lambda, F, r1, r2, r3, r4, r5, k, i);
245 crossed = true;
246 } else {
247 u[k] = popMatrix[i][k];
248 }
249 }
250 if (crosstype == 2) {
251 crossed = true;
252 }
253 }
254
255 // Selection
256 Problem p = new Problem(u, problem);
257 Problem x = new Problem(popMatrix[i], problem);
258 if (Math.abs(p.getFitness()) <= Math.abs(x.getFitness())) {
259 popMatrix[i] = u;
260 }
261 }
262 generation++;
263 }
264 return popMatrix[bestSol()];
265 }

298 private double method(int method, double lambda, double F, int r1, int r2, int r3, int r4, int r5, int k, int i) {
299 switch (method) {
300 case 1: // DE/best/1
301 return popMatrix[bestSol()][k] + F * (popMatrix[r1][k] - popMatrix[r2][k]);
302 case 2: // DE/rand/1
303 return popMatrix[r1][k] + F * (popMatrix[r2][k] - popMatrix[r3][k]);
304 case 3: // DE/rand-to-best/1
305 return popMatrix[i][k] + lambda * (popMatrix[bestSol()][k] - popMatrix[i][k])
306 + F * (popMatrix[r1][k] - popMatrix[r2][k]);
307 case 4: // DE/best/2
308 return popMatrix[bestSol()][k]
309 + F * (popMatrix[r1][k] + popMatrix[r2][k] - popMatrix[r3][k] - popMatrix[r4][k]);
310 case 5: // DE/rand/2
311 return popMatrix[r5][k]
312 + F * (popMatrix[r1][k] + popMatrix[r2][k] - popMatrix[r3][k] - popMatrix[r4][k]);
313 default: // Invalid method
314 return Double.NaN;
315 }
316 }

345 public Particle PSO(int iterations, int numParticles, int dimensions, double c1, double c2) {
346 // Get range of values
347 double range = population.getRange();
348
349 // Initialize the particles
350 Particle[] particles = new Particle[numParticles];
351 for (int i = 0; i < numParticles; i++) {
352 particles[i] = new Particle(popMatrix[i], range, problem);
353 particles[i].setPBest(particles[i]);
354 }
355
356 // Initialize the best particle
357 Particle gBest = particles[0];
358 for (int i = 1; i < numParticles; i++) {
359 if (Math.abs(particles[i].fitness) < gBest.fitness) {
360 gBest = particles[i];
361 }
362 }
363
364 // Runs the main part of the PSO algorithm
365 for (int t = 0; t < iterations; t++) {
366 // Update the states of every particle in the swarm
367 for (int j = 0; j < numParticles; j++) {
368 // Update the velocity and position of the particles
369 for (int k = 0; k < dimensions; k++) {
370 // Calculate the new velocity
371 double addVel = particles[j].velocity
372 + c1 * r.nextDouble() * (particles[j].pBest.solution[k] - particles[j].solution[k])
373 + c2 * r.nextDouble() * (gBest.solution[k] - particles[j].solution[k]);
374 // Add the new velocity to the current position
375 particles[j].solution[k] += addVel;
376 }
377
378 // Update the pBest and fitness of the particle
379 Problem p = new Problem(particles[j].solution, problem);
380 if (Math.abs(p.getFitness()) < Math.abs(particles[j].fitness)) {
381 particles[j].pBest = particles[j];
382 }
383 particles[j].fitness = p.getFitness();
384
385 // Update the gBest
386 if (Math.abs(particles[j].fitness) < gBest.fitness) {
387 gBest = particles[j];
388 }
389 }
390 }
391 return gBest;
392 }
