148 public double griewank() {
149 double sum = 0;

164 public double sine_envelope() {
165 double sum = 0;
166 for (int i = 0; i < values.length - 1; i++) {
167 double top = square(Math.sin(square(values[i]) + square(values[i + 1]) - 0.5));
168 double bottom = square(1 + 0.001 * (square(values[i]) + square(values[i + 1])));

209 public double ackley_two() {
210 double sum = 0;

213 0.2 * Math.sqrt((square(values[i]) + square(values[i + 1])) / 2));

216 sum += 20 + Math.E - (20 / first) - second;
