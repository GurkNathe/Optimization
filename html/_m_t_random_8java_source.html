202 // This is unavoidable because no fields in our instance will

280 int length = buf.length;
281 if (length == 0)
282 throw new IllegalArgumentException("Seed buffer may not be empty");
283 // ---- Begin Mersenne Twister Algorithm ----
284 int i = 1, j = 0, k = (N > length ? N : length);
285 setSeed(MAGIC_SEED);
286 for (; k > 0; k--) {
287 mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >>> 30)) * MAGIC_FACTOR2)) + buf[j] + j;
288 i++;
289 j++;
290 if (i >= N) {
291 mt[0] = mt[N - 1];
292 i = 1;
293 }
294 if (j >= length)
295 j = 0;
296 }

336 protected final synchronized int next(int bits) {
337 // ---- Begin Mersenne Twister Algorithm ----
338 int y, kk;
339 if (mti >= N) { // generate N words at one time
340
341 // In the original C implementation, mti is checked here
342 // to determine if initialisation has occurred; if not
343 // it initialises this instance with DEFAULT_SEED (5489).
344 // This is no longer necessary as initialisation of the
345 // Java instance must result in initialisation occurring
346 // Use the constructor MTRandom(true) to enable backwards
347 // compatible behaviour.
348
349 for (kk = 0; kk < N - M; kk++) {
350 y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
351 mt[kk] = mt[kk + M] ^ (y >>> 1) ^ MAGIC[y & 0x1];
352 }
353 for (; kk < N - 1; kk++) {
354 y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
355 mt[kk] = mt[kk + (M - N)] ^ (y >>> 1) ^ MAGIC[y & 0x1];
356 }
357 y = (mt[N - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
358 mt[N - 1] = mt[M - 1] ^ (y >>> 1) ^ MAGIC[y & 0x1];
359
360 mti = 0;
361 }
362
363 y = mt[mti++];
364
365 // Tempering
366 y ^= (y >>> 11);
367 y ^= (y << 7) & MAGIC_MASK1;
368 y ^= (y << 15) & MAGIC_MASK2;
369 y ^= (y >>> 18);
370 // ---- End Mersenne Twister Algorithm ----
371 return (y >>> (32 - bits));
372 }
